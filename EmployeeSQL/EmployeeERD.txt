Departments
-
dept_no PK VARCHAR(5)
dept_name VARCHAR(30)

Dept_emp
-
emp_no INT FK >- Employees.emp_no
dept_no VARCHAR(30) FK >- Departments.dept_no

Department_managers
-
dept_no VARCHAR(30) FK >- Departments.dept_no
emp_no INT FK >- Employees.emp_no

Employees
-
emp_no PK INT
emp_title_id VARCHAR(30) FK >- Titles.title_id
birth_date DATE
first_name VARCHAR(30)
last_name VARCHAR(30)
sex VARCHAR(1)
hire_date DATE

Salaries
-
emp_no INT FK >- Employees.emp_no
salary INT

Titles
-
title_id PK VARCHAR(5)
title VARCHAR(30)
